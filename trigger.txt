CREATE OR REPLACE FUNCTION partners()
  RETURNS trigger AS
$$
BEGIN
         INSERT INTO partners(col1,col2)
         VALUES(NEW.col1,NEW.col2,NEW.col3);
 
    RETURN NEW;
END;
$$
LANGUAGE 'plpgsql';
